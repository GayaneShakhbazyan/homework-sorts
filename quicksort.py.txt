def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    
    first = arr[0]  
    left = []      
    right = []      

    for x in arr[1:]:  
        if x < first:
            left.append(x)
        else:
            right.append(x)
    return quick_sort(left) + [first] + quick_sort(right)
n = [6, 27, 46, 21, 1]
print(quick_sort(n))
